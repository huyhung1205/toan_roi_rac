1. Tìm kiếm theo chiều rộng (BFS)

  • Cho đồ thị vô hướng $G=(V,E)$, 1 đỉnh nguồn (đỉnh xuất phát) $s$
    • Thuật toán BFS còn được gọi là thuật toán vết dầu loang được xây dựng trên cơ sở hàng đợi (Queue) - vào trước ra trước - FIFO (First In First Out). Nghĩa là, đỉnh được thăm càng sớm sẽ càng sớm trở thành đã duyệt xong và rời khỏi hàng đợi. Nội dung của BFS được tóm tắt như sau: Xuất phát từ một đỉnh nào đó của đồ thị $G$, giả sử là $s$.
    • Đỉnh này xem như đã được thăm-duyệt và gọi là đỉnh mức 0, ta lần lượt xét tất cả các đỉnh kề với nó và được gọi là các đỉnh mức 1, các đỉnh này xem như đã được thăm duyệt, sắp xếp chúng theo một thứ tự nào đó (tương ứng với một hàng đợi). Tương tự, với mỗi đỉnh mức 1 theo thứ tự đã sắp xếp, lần lượt xét các đỉnh kề với nó (chưa được thăm-duyệt), tập các đỉnh kề với các đỉnh mức 1 được gọi là đỉnh mức 2, các đỉnh này xem như đã được thăm-duyệt.
    • Tiếp tục tương tự như vậy cho đến khi tất cả các đỉnh đã được thăm-duyệt.
  • B1: $Q=\emptyset$, enqueue$(Q)$ (nạp $s$ vào $Q$) thì $Q=\{s\}$
  • B2: dequeue($Q$) (lấy $s$ từ $Q$) và kiểm tra $Ke[s]$, enqueue($Q$) ($Q=\{b,e\}$)
  • B3: dequeue($Q$) (lấy $b$ từ $Q$, $Q=\{e\}$) và kiểm tra $Ke[b]$, enqueue($Q$) ($Q=\{e,a,c,k\}$)
  • B4: dequeue($Q$) (lấy $e$ từ $Q$, $Q=\{a,c,k\}$) và kiểm tra $Ke[e]$, enqueue($Q$) ($Q=\{a,c,k\}$)
  • B5: dequeue($Q$) (lấy $a$ từ $Q$, $Q=\{c,k\}$) và kiểm tra $Ke[a]$, enqueue($Q$) ($Q=\{c,k\}$)
  • B6: dequeue($Q$) (lấy $c$ từ $Q$, $Q=\{k\}$) và kiểm tra $Ke[c]$, enqueue($Q$) ($Q=\{k,d,g\}$)
  • B7: dequeue($Q$) (lấy $k$ từ $Q$, $Q=\{d,g\}$) và kiểm tra $Ke[k]$, enqueue($Q$) ($Q=\{d,g\}$)
  • B8: dequeue($Q$) (lấy $d$ từ $Q$, $Q=\{g\}$) và kiểm tra $Ke[d]$, enqueue($Q$) ($Q=\{g\}$)
  • B9: dequeue($Q$) (lấy $g$ từ $Q$, $Q=\emptyset$) và kiểm tra $Ke[g]$, enqueue($Q$) ($Q=\emptyset$)
    • Kết quả thu được của thuật toán BFS xuất phát từ $s$ và được ký hiệu là BFS($s$) là một thứ tự thăm-duyệt các đỉnh trên đồ thị $G$ theo $s$. Một đặc điểm của thuật toán BFS đó là mức của một đỉnh $v$ cho biết độ dài của đường đi ngắn nhất (theo số cạnh) từ $s$ đến $v$ trong đồ thị $G$.

    # Code đầy đủ cho đồ thị kiểu danh sách kề: vẽ đồ thị trước và sau khi duyệt; duyệt mọi đỉnh; sau khi duyệt thì tính đường đi giữa 2 đỉnh bất kì
import networkx as nx
import matplotlib.pyplot as plt
from collections import deque    

def bfs(graph, start):
    """
    param graph: Đồ thị dưới dạng danh sách kề
    param start: Đỉnh bắt đầu tìm kiếm
    """
    color = {v: "white" for v in graph}   # Trắng: chưa thăm
    distance = {v: float("inf") for v in graph}  #K/cach tại mỗi đỉnh ban đầu là inf
    parent = {v: None for v in graph}  # cha của mỗi đỉnh ban đầu là None
    
    color[start] = "gray"   # Đánh dấu đang thăm đỉnh start
    distance[start] = 0     # Khoảng cách từ start đến chính nó
    parent[start] = None    # start là gốc
    Q = deque([start])  # Khởi tạo hàng đợi với đỉnh start
    
    print("BFS duyệt từ đỉnh", start)
    bfs_tree_edges = []  # Lưu trữ các cạnh trong cây BFS để vẽ đồ thị sau khi duyệt

    while Q:
        u = Q.popleft()  # Lấy đỉnh u ra khỏi hàng đợi
        print(f"Đang thăm đỉnh {u}, khoảng cách từ đỉnh gốc: {distance[u]}")

        for v in graph[u]:   # Duyệt các đỉnh kề với u
            if color[v] == "white":  # Nếu v chưa được thăm
                color[v] = "gray"    # Đánh dấu đang thăm
                distance[v] = distance[u] + 1  # Cập nhật khoảng cách
                parent[v] = u         # Ghi nhận cha của v là u
                Q.append(v)       # Đưa v vào hàng đợi
                bfs_tree_edges.append((u, v))  # Lưu cạnh của cây BFS
        color[u] = "black"  # Đánh dấu u đã thăm xong
    return distance, parent, bfs_tree_edges

def build_bfs_tree(bfs_edges):# xây dựng dữ liệu cho đồ thị
    bfs_tree = {u: [] for edge in bfs_edges for u in edge} # xây dựng dữ liệu dạng dict, tạo các key là các đỉnh, value tương ứng là ds rỗng
    """bfs_tree = {}
           for edge in bfs_edges:  # Duyệt qua từng cạnh
               for u in edge:      # Duyệt qua từng đỉnh trong cạnh
                   if u not in bfs_tree:
                       bfs_tree[u] = []  # Thêm đỉnh làm key, gán giá trị ban đầu là danh sách rỗng
    """
    for u, v in bfs_edges:
        bfs_tree[u].append(v)
        bfs_tree[v].append(u)
    return bfs_tree

def get_paths_from_start(parent, start):    #Trả về đường đi từ đỉnh bắt đầu đến tất cả các đỉnh khác dưới dạng dict.
    paths = {}
    for node in parent:
        path = []
        current = node
        while current is not None:
            path.append(current)
            current = parent[current]
        path.reverse()  # Đảo ngược để có đường đi từ start đến node
        paths[node] = path
    return paths

def calculate_distance_bfs_tree(bfs_tree, start, end): #Tính khoảng cách giữa hai đỉnh trên cây BFS
    """
    return: Khoảng cách giữa start và end, hoặc -1 nếu không có đường đi
    """
    if start not in bfs_tree or end not in bfs_tree:
        return -1
    
    visited = {v: False for v in bfs_tree}
    Q = deque([(start, 0)])  # Hàng đợi (đỉnh, khoảng cách)
    while Q:
        current, dist = Q.popleft()
        if current == end:
            return dist
        if not visited[current]:
            visited[current] = True
            for neighbor in bfs_tree[current]:
                if not visited[neighbor]:
                    Q.append((neighbor, dist + 1))
    return -1  # Không tìm thấy đường đi

def draw_graph(graph, title):
    """
    Vẽ đồ thị từ danh sách kề.
    Args: graph (dict): Danh sách kề của đồ thị, title (str): Tiêu đề của đồ thị.
    """
    G = nx.Graph()
    for u in graph:
        for v in graph[u]:
            G.add_edge(u, v)
    pos = nx.spring_layout(G)
    nx.draw(G, pos, with_labels=True, node_color='skyblue', edge_color='black', node_size=500, font_size=12)
    plt.title(title)
    plt.show()

import networkx as nx

def bfs(matrix, start):
    color = ["white"] * n  # Trạng thái của các đỉnh: white (chưa thăm), gray (đang thăm), black (đã thăm)
    distance = [float("inf")] * n  # Khoảng cách từ đỉnh bắt đầu
    parent = [None] * n  # Cha của mỗi đỉnh trong cây BFS
    start -= 1 # Chuyển chỉ số từ 1 thành chỉ số mảng (bắt đầu từ 0)

    color[start] = "gray"
    distance[start] = 0
    Q = deque([start])
    print(f"BFS duyệt từ đỉnh {start + 1}")  # bắt đầu với chỉ số 1
    bfs_tree_edges = []  # Danh sách cạnh của cây BFS
    while Q:
        u = Q.popleft()  # Lấy đỉnh đầu tiên ra khỏi hàng đợi
        print(f"Đang thăm đỉnh {u + 1}, khoảng cách từ đỉnh gốc: {distance[u]}")  # Chuyển chỉ số sang 1-based
        for v in range(n):
            if matrix[u][v] == 1 and color[v] == "white":
                color[v] = "gray"
                distance[v] = distance[u] + 1
                parent[v] = u+1 #giúp chuẩn về chỉ số, do chỉ số của mảng thấp hơn 1 so với chỉ số của đỉnh
                Q.append(v)  # Nạp đỉnh kề vào hàng đợi
                bfs_tree_edges.append((u + 1, v + 1))  # Lưu cạnh với chỉ số từ 1
        color[u] = "black"
    return distance, parent, bfs_tree_edges

def build_bfs_tree(bfs_edges, n):  #Xây dựng cây BFS dưới dạng ma trận kề từ danh sách cạnh.
    bfs_tree = [[0] * n for _ in range(n)]
    for u, v in bfs_edges:
        bfs_tree[u - 1][v - 1] = 1
        bfs_tree[v - 1][u - 1] = 1
    return bfs_tree

def get_paths_from_start(parent, start):  #Trả về đường đi từ đỉnh bắt đầu đến tất cả các đỉnh khác dưới dạng dict.
    """
    return: Dictionary chứa đường đi từ start đến các đỉnh khác.
    """
    paths = {}
    for node in range(len(parent)):
        path = []
        current = node + 1  # Chuyển về 1-based
        while current is not None:
            path.append(current)
            current = parent[current - 1]  # Lấy cha từ danh sách parent (1-based)
        path.reverse()  # Đảo ngược để có đường đi từ start đến node
        paths[node + 1] = path
    return paths

def calculate_distance_bfs_tree(bfs_tree, start, end):  #Tính khoảng cách giữa hai đỉnh trên cây BFS
    """
    return: Khoảng cách giữa start và end, hoặc -1 nếu không có đường đi
    """
    visited = [False] * n
    Q = deque([(start - 1, 0)])
    while Q:
        current, dist = Q.popleft()  # Lấy đỉnh đầu tiên từ hàng đợi
        if current == end - 1:  # Nếu đến đỉnh kết thúc
            return dist
        if not visited[current]:
            visited[current] = True
            for neighbor in range(n):
                if bfs_tree[current][neighbor] == 1 and not visited[neighbor]:
                    Q.append((neighbor, dist + 1))  # Thêm đỉnh kề vào hàng đợi với khoảng cách tăng thêm 1
    return -1  # Không có đường đi giữa start và end

      • Mỗi đỉnh được nạp vào và loại ra khỏi hàng đợi một lần, mỗi thao tác đòi hỏi thời gian $O(1)$. Như vậy tổng thời gian làm việc với hàng đợi là $O(|V|)$.
      • Danh sách kề của mỗi đỉnh được duyệt qua đúng một lần. Tổng độ dài của tất cả các danh sách kề là $O(|E|)$.
      • Cập nhật color, distance, parent, và thêm các cạnh vào danh sách bfs_tree_edges. Những thao tác này đều tốn thời gian $O(1)$ cho mỗi đỉnh hoặc cạnh.
    • Tổng cộng ta có thời gian tính của BFS(s) là $O(|V|+|E|)$, là tuyến tính theo kích thước của danh sách kề biểu diễn đồ thị.

    2. Tìm kiếm theo chiều sâu (DFS)

  • Cho đồ thị vô hướng $G=(V,E)$, 1 đỉnh nguồn (đỉnh xuất phát) $s$

    • Ý tưởng:

  • Bắt đầu tại một đỉnh $v_0$ nào đó, chọn một đỉnh $u$ bất kỳ kề với $v_0$ và lấy nó làm đỉnh duyệt tiếp theo. Cách duyệt tiếp theo được thực hiện tương tự như đối với đỉnh $v_0$ và lúc này coi $u$ là đỉnh bắt đầu (Thủ tục đệ quy).

def dfs(graph, start, visited, parent=None, path=None, depth=0, depths=None):
    if parent is None:
        parent = {v: None for v in graph}
    if depths is None:
        depths = {}
    if path is None:
        path = {v: [] for v in graph}

    visited[start] = True
    depths[start] = depth
    path[start] = path[parent[start]] + [start] if parent[start] is not None else [start]
    print(f"{'  ' * depth}Thăm đỉnh {start}, độ sâu {depth}")
    for v in graph[start]:
        if not visited[v]:
            parent[v] = start
            dfs(graph, v, visited, parent, path, depth + 1, depths)
    return parent, depths, path

def build_dfs_tree(parent):
    dfs_tree = {node: [] for node in parent}
    for child, par in parent.items():
        if par is not None:
            dfs_tree[par].append(child)
            dfs_tree[child].append(par)  # Đảm bảo tính đối xứng
    return dfs_tree

def shortest_distance_dfs(path, node1, node2):
    path1 = path[node1] # Lấy đường đi từ gốc đến hai đỉnh
    path2 = path[node2]
    lca_depth = 0  # Tìm tổ tiên chung gần nhất (LCA)
    while lca_depth < min(len(path1), len(path2)) and path1[lca_depth] == path2[lca_depth]:

    lca_depth -= 1
    return (len(path1) - lca_depth - 1) + (len(path2) - lca_depth - 1) # Tính khoảng cách

    from collections import deque

def bfs(graph, u, color, con):
    con.append(u)

    Q = deque([u])  # Khởi tạo hàng đợi với đỉnh u
    while Q:
        u = Q.popleft()  # Lấy đỉnh current ra khỏi hàng đợi
        for v in graph[u]:  # Duyệt các đỉnh kề với current
            if color[v] == "white":  # Nếu v chưa được thăm
                color[v] = "gray"  # Đánh dấu đang thăm
                Q.append(v)  # Đưa v vào hàng đợi
                con.append(v)  # Thêm v vào thành phần liên thông
        color[u] = "black"  # Đánh dấu current đã thăm xong
    return con
